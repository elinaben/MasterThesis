# ── magick composite builder ───────────────────────────────────────────────────
PANEL_W  <- 500L
LEFT_W   <- 72L
RIGHT_W  <- 80L
TOTAL_W  <- LEFT_W + PANEL_W + RIGHT_W
HEADER_H <- 38L
GAP_H    <- 10L
FS       <- 15L
BG       <- "white"

fit_w <- function(img) {
  h <- as.integer(round(image_info(img)$height * PANEL_W / image_info(img)$width))
  image_resize(img, paste0(PANEL_W, "x", h, "!"))
}

add_border <- function(img, px = 2L) {
  image_border(img, "black", paste0(px, "x", px))
}

make_row <- function(img, kda_label, protein_label) {
  h     <- image_info(img)$height
  left  <- image_blank(LEFT_W, h, color = BG) |>
    image_annotate(kda_label, size = FS - 1, color = "black",
                   font = "sans", gravity = "West", location = "+4+0")
  right <- image_blank(RIGHT_W, h, color = BG) |>
    image_annotate(protein_label, size = FS, color = "black",
                   font = "sans", style = "italic",
                   gravity = "West", location = "+6+0")
  image_append(c(left, img, right), stack = FALSE)
}

make_header <- function(labels) {
  header <- image_blank(TOTAL_W, HEADER_H, color = BG)
  col_w  <- as.integer(PANEL_W / length(labels))
  for (i in seq_along(labels)) {
    x <- LEFT_W + as.integer((i - 1L) * col_w + col_w / 2L) - 28L
    header <- image_annotate(header, labels[i],
                             size = FS - 2, color = "black",
                             font = "sans", weight = 700,
                             gravity = "NorthWest",
                             location = paste0("+", x, "+2"))
  }
  header
}

build_composite <- function(tau_img, act_img,
                            tau_kda, act_kda,
                            tau_label, act_label,
                            lane_labels, title_label) {
  tau_img <- fit_w(tau_img) |> add_border()
  act_img <- fit_w(act_img) |> add_border()
  gap     <- image_blank(TOTAL_W, GAP_H, color = BG)

  title_row <- image_blank(TOTAL_W, 28L, color = BG) |>
    image_annotate(title_label, size = FS + 1, color = "black",
                   font = "sans", weight = 700, gravity = "NorthWest",
                   location = paste0("+", LEFT_W, "+4"))

  comp <- image_append(
    c(title_row, make_header(lane_labels),
      make_row(tau_img, tau_kda, tau_label),
      gap,
      make_row(act_img, act_kda, act_label)),
    stack = TRUE
  ) |> image_border(BG, "8x8")

  comp
}

actin_wb1 <- raw_800_sm |>
  select(Condition, area) |> mutate(WB = "1st WB — panTau blot")

actin_wb2 <- actin_800_2 |>
  left_join(lane_map_2, by = c("lane" = "Lane")) |>
  select(Condition, area = actin) |>
  mutate(WB = "2nd WB — pTau blot",
         Condition = factor(Condition, levels = soma_conditions))

actin_both <- bind_rows(actin_wb1, actin_wb2) |>
  mutate(Condition = factor(as.character(Condition), levels = soma_conditions),
         WB = factor(WB, levels = c("1st WB — panTau blot", "2nd WB — pTau blot")))

ggplot(actin_both, aes(x = Condition, y = area, fill = WB)) +
  geom_col(position = position_dodge(width = 0.75), colour = "black", width = 0.65) +
  scale_fill_manual(values = c("1st WB — panTau blot" = COL_BLUE,
                               "2nd WB — pTau blot"   = COL_GREY)) +
  labs(title = "Soma Actin — 1st vs 2nd WB",
       x = NULL, y = "Band intensity (a.u.)", fill = NULL) +
  theme_classic(base_size = 12) +
  theme(axis.text.x    = element_text(angle = 35, hjust = 1),
        legend.position = "top")
